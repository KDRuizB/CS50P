def convert(fraction):
    while True:
        try:
            # Try to split the fuel
            numerator, denominator = fraction.split("/")
            # make to integer
            new_numerator = int(numerator)
            new_denominator = int(denominator)
            # calc the percentage
            f = new_numerator / new_denominator
            # Check if itÂ´s less then 1
            if f <= 1:
                p = int(f * 100)
                return p
            else:
                fraction = input("Fraction: ")
                pass
        except (ValueError, ZeroDivisionError):
            raise
def gauge(percentage):
    # if percentage is less than 1
    if percentage <= 1:
        return "E"
    # if percentage is more then 99
    elif percentage >= 99:
        return "F"
    # else
    else:
        return str(percentage) + "%"

from fuel import convert, gauge
import pytest

def test_input():
   assert convert("2/4") == (50) and gauge(50) == '50%'
   assert convert("99/100") == (99) and gauge(99) == 'F'
   assert convert("1/100") == (1) and gauge(1) == 'E'

def test_exceptions():
   with pytest.raises(ZeroDivisionError):
      convert("1/0")
   with pytest.raises(ValueError):
      convert("abcd/abdc")
